Composite Shapes

Generic CompositeShape

sheet mySheet +style="dynamic" {
  [@10,10] group myGroup +composition="vertical-stack" {
      text text11 := "1. some text";
      text text12 := "1. some more text";
  }
  [@10,50] group myGroup2 +composition="horizontal-stack" {
      text text21 := "2. some text";
      text text22 := "2. some more text";
  }
  [@10,90] group myGroup3 +composition="horizontal-stack" {
    group mySubGroup1 +composition="vertical-stack" {
      text text31 := "3. some text";
      text text32 := "3. some more text";
    }
    group myGroup +composition="vertical-stack" {
      text text41 := "4. some text";
      text text42 := "4. some more text";
    }
  }
}

Rectangle CompositeShape

sheet mySheet +style="dynamic" {
  [@10,10] group myGroup +composition="vertical-stack" {
    rectangle myHeader +borderWidth=1 +lineColor="red" 
                       +fillColor="rgba(255,255,200,1)" 
                       +composition="vertical-stack:center" {
      text text11 := "1. some text";
      text text12 := "1. some more text";
    }
    rectangle myBody +borderWidth=1 +lineColor="red" 
                 +fillColor="rgba(255,255,200,1)" {
      text text21 := "2. some text";
      text text22 := "2. some more text";
    }
    rectangle myFooter +borderWidth=1 +lineColor="red" 
                 +fillColor="rgba(255,255,200,1)" {
      text text31 := "3. some text";
      text text32 := "3. some more text";
    }
  }
}
